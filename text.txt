<script>
  // Year
  document.getElementById('year').textContent = new Date().getFullYear();

  // Product prices map (used for quote cart totals)
  const productPrices = {
    "Premium Copy Paper": 8.99,
    "HP 201A Black Toner": 74.95,
    "Ergonomic Mesh Chair": 129.00,
    "Pilot G2 Gel Pens (12-pack)": 14.50,
    "Heavy-Duty Shelves": 199.00,
    "General Essentials - Coming Soon": 0
  };

  // Quote cart with quantities and totals
  const quote = [];

  function addToQuote(item) {
    const existing = quote.find(q => q.name === item);
    if (existing) {
      existing.qty++;
    } else {
      quote.push({ name: item, qty: 1, price: productPrices[item] || 0 });
    }
    renderQuote();
  }

  function updateQty(index, newQty) {
    quote[index].qty = Math.max(1, parseInt(newQty) || 1);
    renderQuote();
  }

  function removeItem(index) {
    quote.splice(index, 1);
    renderQuote();
  }

  function clearQuote() {
    quote.length = 0;
    renderQuote();
  }

  function renderQuote() {
    const body = document.getElementById('quoteBody');
    const subtotalEl = document.getElementById('quoteSubtotal');
    const taxEl = document.getElementById('quoteTax');
    const shipEl = document.getElementById('quoteShip');
    const totalEl = document.getElementById('quoteTotal');

    body.innerHTML = '';
    let subtotal = 0;

    quote.forEach((item, i) => {
      const rowSubtotal = item.qty * item.price;
      subtotal += rowSubtotal;
      body.innerHTML += `
        <tr>
          <td>${item.name}</td>
          <td><input type="number" min="1" value="${item.qty}" style="width:60px;" onchange="updateQty(${i}, this.value)"></td>
          <td>$${item.price.toFixed(2)}</td>
          <td>$${rowSubtotal.toFixed(2)}</td>
          <td><button class="btn" onclick="removeItem(${i})">Remove</button></td>
        </tr>
      `;
    });

    const shipping = parseFloat(document.getElementById('shippingEstimate').value) || 0;
    const taxRate = parseFloat(document.getElementById('taxRate').value) || 0;
    const tax = subtotal * (taxRate / 100);
    const total = subtotal + tax + shipping;

    subtotalEl.textContent = subtotal.toFixed(2);
    taxEl.textContent = tax.toFixed(2);
    shipEl.textContent = shipping.toFixed(2);
    totalEl.textContent = total.toFixed(2);
  }

  function sendQuote() {
    const subtotal = document.getElementById('quoteSubtotal').textContent;
    const tax = document.getElementById('quoteTax').textContent;
    const shipping = document.getElementById('quoteShip').textContent;
    const total = document.getElementById('quoteTotal').textContent;

    let bodyText = "Items:\n";
    quote.forEach(item => {
      bodyText += `- ${item.name} x${item.qty} = $${(item.qty * item.price).toFixed(2)}\n`;
    });
    bodyText += `\nSubtotal: $${subtotal}\nTax: $${tax}\nShipping: $${shipping}\nEstimated Total: $${total}`;

    const subject = encodeURIComponent('Quote Request — RackIt AnD StacklT LLC');
    const body = encodeURIComponent(bodyText + "\n\nPlease reply with final pricing and availability.");
    window.location.href = `mailto:hello@rackitandstackit.com?subject=${subject}&body=${body}`;
  }

  // Save/Print Quote (with logo and tagline)
  function printQuote() {
    const subtotal = document.getElementById('quoteSubtotal').textContent;
    const tax = document.getElementById('quoteTax').textContent;
    const shipping = document.getElementById('quoteShip').textContent;
    const total = document.getElementById('quoteTotal').textContent;

    // Hosted logo URL
    const logoUrl = "https://copilot.microsoft.com/th/id/BCO.4f0cb073-3515-4818-aa5b-985217153807.png";

    let content = `
      <div style="text-align:center; margin-bottom:20px;">
        <img src="${logoUrl}" alt="RackIt AnD StacklT LLC Logo" style="max-height:80px;"><br>
        <h2 style="margin:8px 0;">RackIt AnD StacklT LLC — Quote Summary</h2>
        <p style="font-weight:bold; letter-spacing:1px; margin-top:4px;">MAKE YOUR SPACE WORK</p>
      </div>
      <table border="1" cellspacing="0" cellpadding="6" style="border-collapse:collapse; width:100%;">
        <tr><th>Item</th><th>Qty</th><th>Price</th><th>Subtotal</th></tr>
    `;

    quote.forEach(item => {
      content += `<tr>
        <td>${item.name}</td>
        <td>${item.qty}</td>
        <td>$${item.price.toFixed(2)}</td>
        <td>$${(item.qty * item.price).toFixed(2)}</td>
      </tr>`;
    });

    content += `</table>
      <p><strong>Subtotal:</strong> $${subtotal}</p>
      <p><strong>Estimated Tax:</strong> $${tax}</p>
      <p><strong>Shipping:</strong> $${shipping}</p>
      <p><strong>Estimated Total:</strong> $${total}</p>
      <p><em>Generated by RackIt AnD StacklT LLC website</em></p>
    `;

    const w = window.open('', '', 'width=800,height=600');
    w.document.write(content);
    w.document.close();
    w.print();
  }

  // Recalculate totals when shipping/tax inputs change
  document.addEventListener('input', (e) => {
    if (e.target.id === 'shippingEstimate' || e.target.id === 'taxRate') {
      renderQuote();
    }
  });

  // Contact form (mailto)
  document.getElementById('contactForm').addEventListener('submit', (e) => {
    e.preventDefault();
    const fd = new FormData(e.target);
    const subject = encodeURIComponent('New Inquiry — RackIt AnD StacklT LLC');
    const body = encodeURIComponent(
      `Name: ${fd.get('name')}\nEmail: ${fd.get('email')}\n\nMessage:\n${fd.get('message')}`
    );
    window.location.href = `mailto:hello@rackitandstackit.com?subject=${subject}&body=${body}`;
    e.target.reset();
  });
</script>
